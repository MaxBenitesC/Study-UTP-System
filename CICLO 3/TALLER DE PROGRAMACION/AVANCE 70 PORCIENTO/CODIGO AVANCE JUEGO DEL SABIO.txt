import java.util.Scanner;

public class ModoDeJuego {
    private int modoJuego;
    private int numJugadores;

    public void eleccionModoJuego() {
        Scanner teclado = new Scanner(System.in);
        do{
            mostrarMenuPrincipal.mostrar();
            System.out.print("Elige una opciГіn: ");
            modoJuego = teclado.nextInt();
            switch (modoJuego) {
                case 1:
                    System.out.println("Modo Solitario seleccionado.");
                    numJugadores = 1;
                    break;
                case 2:
                    System.out.println("Modo Versus seleccionado.");
                    while (numJugadores != 2 && numJugadores != 3) {
                        System.out.print("Ingrese cantidad de jugadores (2 o 3): ");
                        numJugadores = teclado.nextInt();
                    }
                    break;
                case 9:
                    System.out.println("ВЎAdios!");
                    return;
                default:
                    System.out.println("OpciГіn invГЎlida. Intenta nuevamente.");
            }
        } while (modoJuego != 1 && modoJuego != 2);
    }

    public int getModoJuego() {
        return modoJuego;
    }

    public int getNumJugadores() {
        return numJugadores;
    }

Jugador:

public class Jugador {
    private String nombre;
    private String rol;
    private int ps;
    private int energia;
    private int xp;
    private int racha;
    private boolean eliminado;
    private String nombreAtaque;

    public Jugador(String nombreJugador, String rolJugador) {
        nombre = nombreJugador;
        rol = rolJugador.toLowerCase();
        switch (rol) {
            case "fuego":
                ps = 100;
                energia = 30;
                break;
            case "agua":
                ps = 70;
                energia = 60;
                break;
            default:
                ps = 80;
                energia = 45;
                break;
        }
        xp = 0;
        racha = 0;
        eliminado = false;
    }

    public String NombreAtaque(){
        return nombreAtaque;
    }

    public String getNombre() {
        return nombre;
    }

    public String getRol() {
        return rol;
    }

    public int getPs() {
        return ps;
    }

    public int getEnergia() {
        return energia;
    }

    public int getXp() {
        return xp;
    }

    public boolean estaEliminado() {
        return eliminado;
    }

    public void sumarXp(int valor) {
        xp += valor;
    }

    public void sumarRacha() {
        racha++;
    }

    public void reiniciarRacha() {
        racha = 0;
    }

    public int getRacha() {
        return racha;
    }

    public void curar(int valor) {
        ps += valor;
    }

    public void setRacha(int valor) {
        racha = valor;
    }

    public void recibirDano(int valor) {
        ps -= valor;
        if (ps < 0) {
            ps = 0;
        }
    }
    public void eliminar() {
        eliminado = true;
    }
}
