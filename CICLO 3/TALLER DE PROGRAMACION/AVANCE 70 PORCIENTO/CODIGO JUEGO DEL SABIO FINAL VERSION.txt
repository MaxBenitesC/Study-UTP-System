import java.util.Scanner;

public class ModoDeJuego {
    private int modoJuego;
    private int numJugadores;

    public void eleccionModoJuego() {
        Scanner teclado = new Scanner(System.in);
        do{
            MensajeDurantePartida.mostrarMenuPrincipal();
            System.out.print("Elige una opción: ");
            modoJuego = teclado.nextInt();
            switch (modoJuego) {
                case 1:
                    System.out.println("---------Modo Solitario seleccionado---------");
                    numJugadores = 1;
                    break;
                case 2:
                    System.out.println("---------Modo Versus seleccionado---------");
                    do{
                        System.out.print("Ingrese cantidad de jugadores (2 o 3): ");
                        numJugadores = teclado.nextInt();
                        if(numJugadores != 2 && numJugadores != 3){
                            System.out.println("Cantidad de jugadores inválido! Elegir 2 o 3.");
                        }
                    }while(numJugadores != 2 && numJugadores != 3);
                    break;
                case 9:
                    MensajeDurantePartida.mostrarFinalPartida();
                    return;
                default:
                    System.out.println("Opción inválida. Intenta nuevamente.");
            }
        } while (modoJuego != 1 && modoJuego != 2);
    }

    public int getModoJuego() {
        return modoJuego;
    }

    public int getNumJugadores() {
        return numJugadores;
    }
}

Jugador:

public class Jugador {
    private String nombre;
    private String rol;
    private int ps;
    private int xp;

    public Jugador() {
        xp = 0;
    }

    public void configurarJugador(String nombreJugador, String rolJugador) {
        nombre = nombreJugador;
        rol = rolJugador;
        if (rolJugador.equals("fuego")) {
            ps = 100;
        } else if (rolJugador.equals("agua")) {
            ps = 70;
        } else {
            ps = 90;
        }
    }

    public String getNombre() {
        return nombre;
    }

    public String getRol() {
        return rol;
    }

    public int getPs() {
        return ps;
    }

    public int getXp() {
        return xp;
    }

    public void sumarXp(int valor) {
        xp += valor;
    }

    public void recibirDano(int valor) {
        ps -= valor;
        if (ps < 0) {
            ps = 0;
        }
    }
}

import java.util.Scanner;

public class RegistrodeJugadores {
    public static void registrarJugador(Jugador[] jugadores) {
        Scanner teclado = new Scanner(System.in);
        String nombre;
        String rol = "";
        int opcionRol;
        for (int i = 0; i < jugadores.length; i++) {
            MensajeDurantePartida.mostrarRegistroJugador(i);
            System.out.print("Ingrese su nombre: ");
            nombre = teclado.next();
            do {
                MensajeDurantePartida.mostrarElementosSabio(nombre);
                System.out.print("Opción → ");
                opcionRol = teclado.nextInt();
                switch (opcionRol) {
                    case 1:
                        rol = "fuego";
                        break;
                    case 2:
                        rol = "agua";
                        break;
                    case 3:
                        rol = "tierra";
                        break;
                    default:
                        System.out.println("Opción inválida. Intenta de nuevo.");
                }
            } while (opcionRol < 1 || opcionRol > 3);
            jugadores[i] = new Jugador();
            jugadores[i].configurarJugador(nombre, rol);
        }
    }

    public static void mostrarInfoJugador(Jugador[] jugadores) {
        System.out.println("\n╔══════════════════════════════════╗");
        System.out.println("║       DETALLE DE JUGADORES       ║");

        System.out.println("╠══════════════════════════════════╣");
        for (int i = 0; i < jugadores.length; i++) {

import java.util.Scanner;

public class ModoSolitario {
    Preguntas generadordePreguntas = new Preguntas();
    private int respuestasCorrectas = 0;
    private int respuestasIncorrectas = 0;

    public void jugarModoSolitario(Jugador[] jugador){
        Scanner teclado=new Scanner(System.in);
        char opcSeleccionada;
        for(int turno=0;turno<10;turno++){
            do{
                generadordePreguntas.mostrarPreguntaCompleta(turno, jugador[0], turno);
                System.out.print(" Elige tu respuesta: ");
                opcSeleccionada=teclado.next().toLowerCase().charAt(0);
                if (opcSeleccionada != 'a' && opcSeleccionada != 'b' && opcSeleccionada != 'c' && opcSeleccionada != 'd') {
                    System.out.println("\nOPCIÓN INVÁLIDA. Debes ingresar a, b, c o d.");
                }
            }while (opcSeleccionada != 'a' && opcSeleccionada != 'b' && opcSeleccionada != 'c' && opcSeleccionada != 'd');
            validarRespuesta(jugador[0],opcSeleccionada,turno);
        }
        ResumenPartida.mostrarResumenSolitario(jugador[0], respuestasCorrectas, respuestasIncorrectas);
        int decision;
        do {
            MensajeDurantePartida.mostrarJugarNuevamente();
            System.out.print("Elige una opción: ");
            decision = teclado.nextInt();
            if (decision == 1) {
                jugarModoSolitario(jugador);
            }
        } while (decision != 1 && decision != 2);
    }
    public void validarRespuesta(Jugador jugador,char respuestaUsuario, int turno) {
        if (respuestaUsuario == generadordePreguntas.getRespuesta(turno).charAt(0)) {
            System.out.println("¡Correcto! Ganas 10 XP.");
            jugador.sumarXp(10);
            respuestasCorrectas++;
        } else {
            System.out.println("Incorrecto. Pierdes 5 PS.");
            System.out.println("La respuesta correcta era: " + generadordePreguntas.getRespuesta(turno));
            jugador.recibirDano(5);
            respuestasIncorrectas++;
        }
    }
}

import java.util.Scanner;

public class ModoVersus {
    Preguntas generadordePreguntas = new Preguntas();

    public void jugarModoVersus(Jugador[] jugador, int cantidadJugadores){
        int[] respuestasCorrectas = new int[cantidadJugadores];
        int[] respuestasIncorrectas = new int[cantidadJugadores];
        Scanner teclado=new Scanner(System.in);
        char opcSeleccionada;
        for(int turno=0;turno<10;turno++){
            for(int i =0;i<cantidadJugadores;i++){
                if (jugador[i].getPs() <= 0) {
                    System.out.println("El jugador " + jugador[i].getNombre() + " se quedó sin PS y fue eliminado.");
                    continue;
                }
                int numPregunta=turno*cantidadJugadores+i;
                do{
                    generadordePreguntas.mostrarPreguntaCompleta(numPregunta, jugador[i], turno);
                    System.out.print(" Elige tu respuesta: ");
                    opcSeleccionada=teclado.next().toLowerCase().charAt(0);
                    if (opcSeleccionada != 'a' && opcSeleccionada != 'b' && opcSeleccionada != 'c' && opcSeleccionada != 'd') {
                        System.out.println("\nOPCIÓN INVÁLIDA. Debes ingresar a, b, c o d.");
                    }
                }while (opcSeleccionada != 'a' && opcSeleccionada != 'b' && opcSeleccionada != 'c' && opcSeleccionada != 'd');
                validarRespuesta(jugador, i, opcSeleccionada, numPregunta, respuestasCorrectas, respuestasIncorrectas, cantidadJugadores);
            }
        }
        ResumenPartida.mostrarResumenVersus(jugador, respuestasCorrectas, respuestasIncorrectas);
        int decision;
        do {
            MensajeDurantePartida.mostrarJugarNuevamente();
            System.out.print("Elige una opción: ");
            decision = teclado.nextInt();
            if (decision == 1) {
                jugarModoVersus(jugador,cantidadJugadores);
            }
        } while (decision != 1 && decision != 2);
    }

    public void validarRespuesta(Jugador[] jugadores, int atacante, char respuestaUsuario, int turno, int[] respuestasCorrectas, int[] respuestasIncorrectas, int cantidadJugadores) {
        Scanner teclado = new Scanner(System.in);
        if (respuestaUsuario == generadordePreguntas.getRespuesta(turno).charAt(0)) {
            System.out.println("¡Correcto! Ganas 10 XP.");
            jugadores[atacante].sumarXp(10);
            respuestasCorrectas[atacante]++;

            if (cantidadJugadores == 2) {
                int victima = (atacante == 0) ? 1 : 0;
                jugadores[victima].recibirDano(10);
                System.out.println("Atacaste a " + jugadores[victima].getNombre() + " y le quitaste 10 PS.");
                if (jugadores[victima].getPs() <= 0) {
                    System.out.println("Jugador " + jugadores[victima].getNombre() + " ha sido eliminado del juego!");
                }
            } else if (cantidadJugadores == 3) {
                int rivalElegido;
                do {
                    System.out.println("¿A quién deseas atacar?");
                    for (int i = 0; i < cantidadJugadores; i++) {
                        if (i != atacante && jugadores[i].getPs() > 0) {
                            System.out.println("[" + (i + 1) + "] " + jugadores[i].getNombre());
                        }
                    }
                    System.out.print("Elige el número del jugador: ");
                    rivalElegido = teclado.nextInt() - 1;
                    if (rivalElegido == atacante || rivalElegido < 0 || rivalElegido >= cantidadJugadores || jugadores[rivalElegido].getPs() <= 0) {
                        System.out.println("Opción inválida. Debes elegir un jugador de la lista.");
                    }
                } while (rivalElegido == atacante || rivalElegido < 0 || rivalElegido >= cantidadJugadores || jugadores[rivalElegido].getPs() <= 0);
                jugadores[rivalElegido].recibirDano(10);
                System.out.println("Atacaste a " + jugadores[rivalElegido].getNombre() + " y le quitaste 10 PS.");
                if (jugadores[rivalElegido].getPs() <= 0) {
                    System.out.println("Jugador " + jugadores[rivalElegido].getNombre() + " ha sido eliminado del juego!");
                }
            }
        } else {
            System.out.println("Incorrecto. Pierdes 5 PS.");
            System.out.println("La respuesta correcta era: " + generadordePreguntas.getRespuesta(turno));
            jugadores[atacante].recibirDano(5);
            respuestasIncorrectas[atacante]++;
            if (jugadores[atacante].getPs() <= 0) {
                System.out.println("Jugador " + jugadores[atacante].getNombre() + " ha sido eliminado del juego!");
            }
        }
    }
}

public class ResumenPartida {
    public static void mostrarResumenSolitario(Jugador jugador, int respuestasCorrectas, int respuestasIncorrectas) {
        System.out.println("\n═══════════════════════════════════════════════");
        System.out.println("\t\t\tRESUMEN DE TU PARTIDA");
        System.out.println("═══════════════════════════════════════════════");
        System.out.printf("Jugador: %s\n", jugador.getNombre());
        System.out.printf("Rol: %s\n", jugador.getRol().toUpperCase());
        System.out.println("-----------------------------------------------");
        System.out.printf("%-25s %5d\n", "Respuestas Correctas:", respuestasCorrectas);
        System.out.printf("%-25s %5d\n", "Respuestas Incorrectas:", respuestasIncorrectas);
        System.out.printf("%-25s %5d\n", "XP Acumulado:", jugador.getXp());
        System.out.printf("%-25s %5d\n", "PS Final:", jugador.getPs());
        System.out.println("-----------------------------------------------");
        if (jugador.getXp() >= 40) {
            System.out.println("\n═══════════════════════════════════════════════════════");
            System.out.println("Felicidades, has demostrado ser un gran Sabio del " + jugador.getRol());
        } else {
            System.out.println("Aún te falta mucho por mejorar.");
        }
        System.out.println("═══════════════════════════════════════════════════════");
    }

    public static void mostrarResumenVersus(Jugador[] jugadores, int[] correctas, int[] incorrectas) {
        System.out.println("\n╔════════════════════════════════════════════════════════════════════════════════════╗");
        System.out.println("║                                RESUMEN FINAL DE PARTIDA                            ║");
        System.out.println("╠════════════╦════════════╦═════════════════╦════════════════════╦════════╦══════════╣");
        System.out.println("║ JUGADOR    ║ ROL        ║ RESP. CORRECTAS ║ RESP. INCORRECTAS  ║ XP     ║ PS FINAL ║");
        System.out.println("╠════════════╬════════════╬═════════════════╬════════════════════╬════════╬══════════╣");
        int maxCorrectas = -1;
        String ganador = "";
        for (int i = 0; i < jugadores.length; i++) {
            System.out.printf("║ %-10s ║ %-10s ║ %-15d ║ %-18d ║ %-6d ║ %-8d ║\n", jugadores[i].getNombre(), jugadores[i].getRol().toUpperCase(), correctas[i], incorrectas[i], jugadores[i].getXp(), jugadores[i].getPs());
            if (correctas[i] > maxCorrectas) {
                maxCorrectas = correctas[i];
                ganador = jugadores[i].getNombre();
            }
        }
        System.out.println("╚════════════╩════════════╩═════════════════╩════════════════════╩════════╩══════════╝");
        System.out.printf("El ganador es %s con %d respuestas correctas!\n", ganador, maxCorrectas);
    }
}
